                    openEditModal(id);
                });
            });
            
            document.querySelectorAll('.action-btn.delete').forEach(btn => {
                btn.addEventListener('click', (e) => {
                    e.stopPropagation();
                    const id = parseInt(btn.getAttribute('data-id'));
                    deleteCompany(id);
                });
            });
        }

        // Filter companies
        function filterCompanies() {
            const searchTerm = searchInput.value.toLowerCase();
            const selectedRegion = regionFilter.value;
            
            let filtered = companies;
            
            if (searchTerm) {
                filtered = filtered.filter(company => 
                    company.nome.toLowerCase().includes(searchTerm) ||
                    company.especialidade.toLowerCase().includes(searchTerm)
                );
            }
            
            if (selectedRegion !== 'todas') {
                filtered = filtered.filter(company => company.regiao === selectedRegion);
            }
            
            renderCompanies(filtered);
            updateStats(filtered.length);
        }

        // Update stats
        function updateStats(count = companies.length) {
            totalCompanies.textContent = count;
        }

        // Open add modal
        function openAddModal() {
            editingCompanyId = null;
            modalTitle.textContent = 'Nova Empresa';
            saveBtn.textContent = 'Adicionar Empresa';
            companyForm.reset();
            modalOverlay.classList.add('active');
        }

        // Open edit modal
        function openEditModal(id) {
            const company = companies.find(c => c.id === id);
            if (!company) return;
            
            editingCompanyId = id;
            modalTitle.textContent = 'Editar Empresa';
            saveBtn.textContent = 'Atualizar Empresa';
            
            document.getElementById('companyId').value = company.id;
            document.getElementById('nome').value = company.nome;
            document.getElementById('endereco').value = company.endereco;
            document.getElementById('telefone').value = company.telefone;
            document.getElementById('regiao').value = company.regiao;
            document.getElementById('horario').value = company.horario;
            document.getElementById('avaliacao').value = company.avaliacao;
            document.getElementById('especialidade').value = company.especialidade;
            
            modalOverlay.classList.add('active');
        }

        // Close modal
        function closeModalFunc() {
            modalOverlay.classList.remove('active');
        }

        // Save company
        function saveCompany() {
            const formData = new FormData(companyForm);
            const companyData = {
                nome: document.getElementById('nome').value,
                endereco: document.getElementById('endereco').value,
                telefone: document.getElementById('telefone').value,
                regiao: document.getElementById('regiao').value,
                horario: document.getElementById('horario').value,
                avaliacao: parseFloat(document.getElementById('avaliacao').value),
                especialidade: document.getElementById('especialidade').value
            };
            
            if (editingCompanyId) {
                // Update existing company
                const index = companies.findIndex(c => c.id === editingCompanyId);
                if (index !== -1) {
                    companies[index] = { ...companyData, id: editingCompanyId };
                }
            } else {
                // Add new company
                const newId = companies.length > 0 ? Math.max(...companies.map(c => c.id)) + 1 : 1;
                companies.push({ ...companyData, id: newId });
            }
            
            renderCompanies();
            updateStats();
            closeModalFunc();
        }

        // Delete company
        function deleteCompany(id) {
            if (confirm('Tem certeza que deseja excluir esta empresa?')) {
                companies = companies.filter(company => company.id !== id);
                renderCompanies();
                updateStats();
            }
        }

        // Initialize the app
        document.addEventListener('DOMContentLoaded', init);
    </script>
